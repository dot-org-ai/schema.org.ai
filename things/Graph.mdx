# Graph

A graph structure representing nodes (vertices) and edges (relationships) that encode knowledge, relationships, and processes in a structured, queryable format. Graphs enable semantic representation of complex domains, support graph-based reasoning, and integrate with GraphDL and MDXLD for Business-as-Code applications.

## Types

The following types extend Graph:

- **KnowledgeGraph**: Semantic knowledge representation graphs
- **PropertyGraph**: Graphs with properties on nodes and edges
- **SemanticGraph**: RDF-based semantic networks
- **ProcessGraph**: Business process and workflow graphs

## Examples

### Example 1: Business Knowledge Graph

```typescript
import { $ } from 'sdk.do'

const businessKG = {
  $type: 'Graph',
  $id: 'https://graph.org.ai/business-knowledge',
  name: 'Business Domain Knowledge Graph',
  description: 'Knowledge graph representing business entities, relationships, and processes across the .do platform ecosystem',
  digital: 1.0,
  graphType: 'knowledge',
  queryLanguage: 'SPARQL',
  schema: 'https://schema.org.ai',
  namespace: 'https://graph.org.ai/',
  format: 'JSON-LD',
  nodes: [
    {
      $type: 'Organization',
      $id: 'https://platform.do',
      name: '.do Platform',
      industry: {
        $type: 'Industry',
        naicsCode: '511210',
        name: 'Software Publishers'
      }
    },
    {
      $type: 'Person',
      $id: 'https://people.do/nathan',
      name: 'Nathan Clevenger',
      jobTitle: 'CEO',
      worksFor: 'https://platform.do'
    },
    {
      $type: 'Product',
      $id: 'https://products.do/sdk',
      name: 'Business-as-Code SDK',
      manufacturer: 'https://platform.do'
    }
  ],
  edges: [
    {
      $type: 'Edge',
      subject: 'https://people.do/nathan',
      predicate: 'worksFor',
      object: 'https://platform.do',
      edgeType: 'employment'
    },
    {
      $type: 'Edge',
      subject: 'https://platform.do',
      predicate: 'produces',
      object: 'https://products.do/sdk',
      edgeType: 'production'
    },
    {
      $type: 'Edge',
      subject: 'https://platform.do',
      predicate: 'belongsToIndustry',
      object: 'https://industries.org.ai/511210',
      edgeType: 'classification'
    }
  ],
  nodeCount: 1247,
  edgeCount: 3891,
  tripleStore: 'https://api.platform.do/sparql',
  version: '2.1.0',
  creator: {
    $type: 'Organization',
    name: '.do Platform Team',
    url: 'https://platform.do'
  }
}

await $.Graph.create(businessKG)

// Query the graph using SPARQL
const query = `
  PREFIX schema: <https://schema.org.ai/>
  SELECT ?person ?title ?company
  WHERE {
    ?person a schema:Person ;
            schema:jobTitle ?title ;
            schema:worksFor ?company .
    ?company a schema:Organization .
  }
`

const results = await $.Graph.query(businessKG.$id, query)
```

## Usage in Business-as-Code

```typescript
import { $ } from 'sdk.do'
import type { Graph, Node, Edge } from 'schema.org.ai'

// Create a graph
const graph: Graph = {
  $type: 'Graph',
  $id: 'https://graph.org.ai/my-graph',
  name: 'My Knowledge Graph',
  graphType: 'property',
  queryLanguage: 'GraphQL',
  digital: 1.0
}

await $.Graph.create(graph)

// Add nodes to the graph
await $.Graph.addNode(graph.$id, {
  $type: 'Organization',
  $id: 'https://example.com',
  name: 'Example Company'
})

// Add edges between nodes
await $.Graph.addEdge(graph.$id, {
  subject: 'https://person.example.com',
  predicate: 'worksFor',
  object: 'https://example.com'
})

// Query the graph
const results = await $.Graph.query(graph.$id, `
  query {
    organizations {
      name
      employees {
        name
        skills
      }
    }
  }
`)

// Export graph in different formats
const jsonLD = await $.Graph.export(graph.$id, 'json-ld')
const turtle = await $.Graph.export(graph.$id, 'turtle')
```
