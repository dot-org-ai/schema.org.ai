## Usage in Business-as-Code

```typescript
import { $ } from 'sdk.do'
import type { Property } from 'schema.org.ai'

// Create a Property
const item: Property = {
  $type: 'Property',
  name: 'Example Property'
}

// Use semantic patterns
await $.Property.create(item)
const result = await $.Property.get('item-id')
```
